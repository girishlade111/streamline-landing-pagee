### 1.2 AnimationWrapper Component
Create `components/animation-wrapper.tsx` - Reusable wrapper for common animations:

```typescript
'use client';

import { motion, MotionProps, Variants } from 'framer-motion';
import { cn } from '@/lib/utils';
import { useReducedMotion } from '@/hooks/use-reduced-motion';
import { ANIMATION_VARIANTS } from '@/lib/motion-config';

interface AnimationWrapperProps extends MotionProps {
  children: React.ReactNode;
  variant?: keyof typeof ANIMATION_VARIANTS;
  className?: string;
  as?: keyof JSX.IntrinsicElements;
  delay?: number;
  duration?: number;
  once?: boolean;
  customVariants?: Variants;
}

export function AnimationWrapper({
  children,
  variant = 'fade',
  className,
  as = 'div',
  delay = 0,
  duration,
  once = true,
  customVariants,
  ...props
}: AnimationWrapperProps) {
  const prefersReducedMotion = useReducedMotion();
  const baseVariants = customVariants || ANIMATION_VARIANTS[variant];
  
  // If user prefers reduced motion, disable animations
  const motionProps = prefersReducedMotion
    ? {
        initial: false,
        animate: false,
        exit: false,
        transition: { duration: 0 },
      }
    : {
        initial: 'initial',
        animate: 'animate',
        exit: 'exit',
        variants: baseVariants,
        custom: { delay, duration },
      };

  return (
    <motion.div
      as={as}
      className={cn(className)}
      {...motionProps}
      {...props}
      style={{
        willChange: 'transform, opacity',
        ...props.style,
      }}
    >
      {children}
    </motion.div>
  );
}

// Specialized wrappers for common use cases
export const PageTransition = ({ children }: { children: React.ReactNode }) => (
  <AnimationWrapper variant="page" className="w-full">
    {children}
  </AnimationWrapper>
);

export const StaggerContainer = ({ children }: { children: React.ReactNode }) => (
  <motion.div
    variants={ANIMATION_VARIANTS.staggerContainer}
    initial="initial"
    whileInView="animate"
    viewport={{ once: true, amount: 0.2 }}
    className="w-full"
  >
    {children}
  </motion.div>
);

export const StaggerItem = ({ children, delay = 0 }: { children: React.ReactNode; delay?: number }) => (
  <motion.div
    variants={ANIMATION_VARIANTS.staggerItem}
    custom={delay}
    className="w-full"
    style={{ willChange: 'transform, opacity' }}
  >
    {children}
  </motion.div>
);
```

### 1.3 Reduced Motion Hook
Create `hooks/use-reduced-motion.ts` - Respects user accessibility preferences:

```typescript
'use client';

import { useState, useEffect } from 'react';

export function useReducedMotion() {
  const [prefersReducedMotion, setPrefersReducedMotion] = useState(false);

  useEffect(() => {
    const mediaQuery = window.matchMedia('(prefers-reduced-motion: reduce)');
    setPrefersReducedMotion(mediaQuery.matches);

    const handleChange = (event: MediaQueryListEvent) => {
      setPrefersReducedMotion(event.matches);
    };

    mediaQuery.addEventListener('change', handleChange);
    return () => mediaQuery.removeEventListener('change', handleChange);
  }, []);

  return prefersReducedMotion;
}
```

import { motion, useScroll, useTransform } from 'framer-motion';
import { useRef } from 'react';

interface ParallaxSectionProps {
  children: React.ReactNode;

  className?: string;

export function ParallaxSection({ 
  children, 
  speed = 0.5, 
  className = ''

  const { scrollYProgress } = useScroll({
    target: ref,
    offset: ['start end', 'end start'],

  return (
    <motion.div
      ref={ref}
      style={{ y, opacity, willChange: 'transform, opacity' }}
      className={className}
    >
      {children}
    </motion.div>
  );
}

### 3.3 Reveal on Scroll Component
Create `components/reveal-on-scroll.tsx`:

```typescript
'use client';

import { motion, useScroll, useTransform } from 'framer-motion';
import { useRef } from 'react';

interface RevealOnScrollProps {
  children: React.ReactNode;
  className?: string;
  direction?: 'up' | 'down' | 'left' | 'right';
}

export function RevealOnScroll({ 
  children, 
  className = '',
  direction = 'up' 
}: RevealOnScrollProps) {
  const ref = useRef<HTMLDivElement>(null);
  const { scrollYProgress } = useScroll({
    target: ref,
    offset: ['start 0.8', 'end 0.2'],
  });

  const directionMap = {
    up: { y: [-20, 0], x: [0, 0] },
    down: { y: [20, 0], x: [0, 0] },
    left: { x: [-20, 0], y: [0, 0] },
    right: { x: [20, 0], y: [0, 0] },
  };

  const y = useTransform(scrollYProgress, [0, 1], directionMap[direction].y);
  const x = useTransform(scrollYProgress, [0, 1], directionMap[direction].x);
  const opacity = useTransform(scrollYProgress, [0, 0.5, 1], [0, 0.5, 1]);

  return (
    <motion.div
      ref={ref}
      style={{ x, y, opacity, willChange: 'transform, opacity' }}
      className={className}
    >
      {children}
    </motion.div>
  );
}
```

export function LazyAnimation({ children, className }: { children: React.ReactNode; className?: string }) {

### 6.3 Optimize for Mobile
```typescript
// hooks/use-mobile-performance.ts
'use client';

import { useState, useEffect } from 'react';

export function useMobilePerformance() {
  const [isMobile, setIsMobile] = useState(false);
  const [reducedMotion, setReducedMotion] = useState(false);

  useEffect(() => {
    const checkMobile = () => setIsMobile(window.innerWidth < 768);
    const checkMotion = () => 
      setReducedMotion(window.matchMedia('(prefers-reduced-motion: reduce)').matches);

    checkMobile();
    checkMotion();

    window.addEventListener('resize', checkMobile);
    window.matchMedia('(prefers-reduced-motion: reduce)').addEventListener('change', checkMotion);

    return () => {
      window.removeEventListener('resize', checkMobile);
      window.matchMedia('(prefers-reduced-motion: reduce)').removeEventListener('change', checkMotion);
    };
  }, []);

  return { isMobile, reducedMotion };
}
```

## Key Principles

1. **Consistency**: All animations use shared spring/damping values
2. **Performance**: GPU-accelerated properties only (transform, opacity)
3. **Accessibility**: Respects `prefers-reduced-motion`
4. **Maintainability**: Central config file for easy updates
5. **Reusability**: Wrapper components for common patterns
6. **User Experience**: Smooth, polished, never distracting
